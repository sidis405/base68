fastcgi_cache_path /etc/nginx/cache levels=1:2 keys_zone=FUNTADEV:100m inactive=60m;
fastcgi_cache_key "$scheme$request_method$host$request_uri";

server {
        listen 80 default;
        root /var/www/web.funta/public;
        index index.php index.html index.htm;
        server_name web.funtadev.com;


    #hide nginx info
        server_tokens off;

    #Mime
        include mime.types;

         gzip on;
        gzip_min_length 100;
        gzip_comp_level 3;
        gzip_disable "msie6";
        gzip_vary on;
        gzip_proxied any;
        gzip_buffers 4 16k;
        gzip_http_version 1.1;

     #types that should gzip
        gzip_types application/ecmascript;
        gzip_types application/javascript;
        gzip_types application/json;
        gzip_types application/pdf;
        gzip_types application/postscript;
        gzip_types application/x-javascript;
        gzip_types application/svg+xml;
        gzip_types image/svg+xml;
        gzip_types text/css;
        gzip_types text/csv;
        gzip_types text/javascript;
        gzip_types text/plain;
        gzip_types text/xml;
        gzip_types image/bmp;
        gzip_types image/x-icon;
        gzip_types image/jpeg;
        gzip_types image/jpg;
        gzip_types image/pjpeg;
        gzip_types image/pjpg;


            # Media: images, icons, video, audio, HTC
        location ~* \.(?:jpg|jpeg|gif|png|ico|cur|gz|svg|svgz|mp4|ogg|ogv|webm|htc)$ {
          expires 1M;
          access_log off;
          add_header Cache-Control "public";
        }

	    add_header X-Cache $upstream_cache_status;


        # CSS and Javascript
        location ~* \.(?:css|js)$ {
          expires 1y;
          access_log off;
          add_header Cache-Control "public";
        }


        location / {
                # First attempt to serve request as file, then
                # as directory, then fall back to displaying a 404.
                try_files $uri $uri/ /index.php?$query_string;
                # Uncomment to enable naxsi on this location
                # include /etc/nginx/naxsi.rules
        }
        location ~ \.php$ {

		try_files $uri =404;
		fastcgi_split_path_info ^(.+\.php)(/.+)$;
        	fastcgi_pass unix:/var/run/php/php7.1-fpm.sock;
	        fastcgi_index index.php;
	        fastcgi_cache FUNTADEV;
        	fastcgi_cache_valid 200 1m;
	        fastcgi_cache_bypass $no_cache;
	        fastcgi_no_cache $no_cache;
        	fastcgi_cache_use_stale updating error timeout invalid_header http_500;
	        fastcgi_cache_lock on;
	        fastcgi_ignore_headers Cache-Control Expires Set-Cookie;
		fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
	        include fastcgi_params;

        }
	#Cache everything by default
    set $no_cache 0;

    #Don't cache POST requests
    if ($request_method = POST)
    {
        set $no_cache 1;
    }

    #Don't cache if the URL contains a query string
    if ($query_string != "")
    {
        set $no_cache 1;
    }

    #Don't cache the following URLs
    if ($request_uri ~* "/(cp/)")
    {
        set $no_cache 1;
    }

    #Don't cache if there is a cookie called PHPSESSID
    if ($http_cookie = "PHPSESSID")
    {
        set $no_cache 1;
    }
}
